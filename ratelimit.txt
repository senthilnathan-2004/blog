1. First, install the necessary packages:

npm install express express-rate-limit express-slow-down



2. Express Code with Rate Limiting + Throttling:

//  Rate Limiter: Max 5 requests per 1 minute per IP
const loginLimiter = rateLimit({
  windowMs: 60 * 1000,   // 1 minute
  max: 5,               // limit each IP to 5 requests per windowMs
  message: "Too many login attempts. Please try again later.",
  standardHeaders: true,
  legacyHeaders: false,
});

//  Slow Down: Delay each request after 3 attempts
const loginSpeedLimiter = slowDown({
  windowMs: 60 * 1000,     // 1 minute
  delayAfter: 3,          // allow 3 requests at full speed
  delayMs: 1000,         // 1 second delay per request after limit
});

//  Apply both middlewares to login route
app.post('/login', loginLimiter, loginSpeedLimiter, (req, res) => {
  // Your login logic here
  res.send("Login attempt handled");
});
